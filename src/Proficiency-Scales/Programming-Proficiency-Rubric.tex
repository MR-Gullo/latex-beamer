\begin{multicols}{2}

%% EMERGING
\begin{tcolorbox}[profbox, colback=emerging!15, colframe=emerging!60]
{\large\bfseries\sffamily EMERGING}\\[2pt]
\small\itshape Beginning to understand basic programming concepts with significant guidance
\tcblower
\small
\begin{itemize}
\item Writes very basic syntax with teacher support and examples
\item Recognizes simple programming structures when shown but struggles to create them independently
\item Completes guided exercises by copying and modifying provided code
\item Submits work more than 3 days late, uses incorrect file formats, leaves work incomplete
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% DEVELOPING
\begin{tcolorbox}[profbox, colback=developing!15, colframe=developing!60]
{\large\bfseries\sffamily DEVELOPING}\\[2pt]
\small\itshape Uses basic control structures with some guidance needed
\tcblower
\small
\begin{itemize}
\item Uses basic control structures (loops, conditionals) in simple programs
\item Solves straightforward problems similar to classroom examples
\item Begins combining multiple control structures with teacher guidance
\item Submits 1--2 days late or requests last-minute extensions; completes most work
\end{itemize}
\end{tcolorbox}

\columnbreak

%% PROFICIENT
\begin{tcolorbox}[profbox, colback=proficient!15, colframe=proficient!60]
{\large\bfseries\sffamily PROFICIENT}\\[2pt]
\small\itshape Independently solves familiar abstract problems
\tcblower
\small
\begin{itemize}
\item Combines multiple control structures to solve problems seen in class
\item Applies learned patterns to solve abstract multistep problems
\item Debugs own code using systematic approaches
\item Submits on time with proper formatting and communicates about potential delays
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% EXTENDING
\begin{tcolorbox}[profbox, colback=extending!15, colframe=extending!60]
{\large\bfseries\sffamily EXTENDING}\\[2pt]
\small\itshape Solves novel abstract problems independently and creatively
\tcblower
\small
\begin{itemize}
\item Solves abstract problems not previously encountered in class
\item Creates original solutions by combining concepts in new ways
\item Analyzes problems to determine appropriate structures and approaches
\item Submits quality work ahead of deadlines with clear communication
\end{itemize}
\end{tcolorbox}
