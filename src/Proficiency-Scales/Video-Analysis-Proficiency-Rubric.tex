%% EMERGING
\begin{tcolorbox}[profbox, colback=emerging!15, colframe=emerging!60,
  title={\colorbox{emerging!80}{\strut\hspace{2pt}EMERGING\hspace{2pt}}}]
\small\itshape Beginning awareness of Newton's Laws and force concepts
\tcblower
\small
\begin{itemize}
\item Submits work more than 3 days late, uses incorrect file formats
\item Identifies basic forces with teacher support but struggles with free-body diagrams
\item Omits most textbook references (page numbers, equation numbers)
\item Participates inconsistently in group work, missing meetings
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% DEVELOPING
\begin{tcolorbox}[profbox, colback=developing!15, colframe=developing!60,
  title={\colorbox{developing!80}{\strut\hspace{2pt}DEVELOPING\hspace{2pt}}}]
\small\itshape Growing understanding with occasional guidance needed
\tcblower
\small
\begin{itemize}
\item Submits 1--2 days late or requests last-minute extensions
\item Applies Newton's Second Law in basic scenarios; creates simple free-body diagrams
\item Includes some textbook references but misses equation numbers or page citations
\item Contributes to group discussions, though coordination needs improvement
\end{itemize}
\end{tcolorbox}

\columnbreak

%% PROFICIENT
\begin{tcolorbox}[profbox, colback=proficient!15, colframe=proficient!60,
  title={\colorbox{proficient!80}{\strut\hspace{2pt}PROFICIENT\hspace{2pt}}}]
\small\itshape Works independently with solid conceptual understanding
\tcblower
\small
\begin{itemize}
\item Submits on time with proper formatting and communicates about delays
\item Creates accurate free-body diagrams, applies all three Newton's Laws, identifies action-reaction pairs
\item Consistently cites textbook pages and equation numbers with complete step-by-step solutions
\item Actively fulfills role responsibilities, coordinates effectively with team
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% EXTENDING
\begin{tcolorbox}[profbox, colback=extending!15, colframe=extending!60,
  title={\colorbox{extending!80}{\strut\hspace{2pt}EXTENDING\hspace{2pt}}}]
\small\itshape Sophisticated understanding with real-world applications
\tcblower
\small
\begin{itemize}
\item Submits quality work ahead of deadlines with clear communication
\item Synthesizes multiple force concepts for complex motion scenarios; evaluates alternative approaches
\item Produces exceptionally detailed citations with context; creates professional diagrams
\item Leads role coordination, facilitates peer understanding, ensures integrated delivery
\end{itemize}
\end{tcolorbox}
