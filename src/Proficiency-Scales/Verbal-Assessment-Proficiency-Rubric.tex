\begin{multicols}{2}

%% EMERGING
\begin{tcolorbox}[profbox, colback=emerging!15, colframe=emerging!60]
{\large\bfseries\sffamily EMERGING}\\[2pt]
\small\itshape Beginning understanding, needs significant teacher support
\tcblower
\small
\begin{itemize}
\item Gives very short answers with help from teacher prompts and questions
\item Uses simple words but may mix up basic terms and subject concepts
\item Shows some recognition of familiar problems but cannot explain the steps to solve them
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% DEVELOPING
\begin{tcolorbox}[profbox, colback=developing!15, colframe=developing!60]
{\large\bfseries\sffamily DEVELOPING}\\[2pt]
\small\itshape Growing understanding, participates with some independence
\tcblower
\small
\begin{itemize}
\item Answers questions with basic explanations but needs occasional teacher guidance
\item Uses correct vocabulary most of the time and can define important subject terms
\item Explains simple problem-solving steps but may struggle with more complex connections
\end{itemize}
\end{tcolorbox}

\columnbreak

%% PROFICIENT
\begin{tcolorbox}[profbox, colback=proficient!15, colframe=proficient!60]
{\large\bfseries\sffamily PROFICIENT}\\[2pt]
\small\itshape Solid understanding, communicates knowledge effectively
\tcblower
\small
\begin{itemize}
\item Provides complete and accurate explanations without teacher prompting
\item Uses technical vocabulary correctly and can explain terms to others
\item Demonstrates clear problem-solving strategies and walks through solutions step by step
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% EXTENDING
\begin{tcolorbox}[profbox, colback=extending!15, colframe=extending!60]
{\large\bfseries\sffamily EXTENDING}\\[2pt]
\small\itshape Deep understanding, articulates complex concepts with confidence
\tcblower
\small
\begin{itemize}
\item Explains difficult concepts in multiple ways and helps other students understand
\item Uses advanced vocabulary appropriately and creates clear analogies for complex ideas
\item Makes connections between different concepts and applies learning to real-world situations
\end{itemize}
\end{tcolorbox}

\end{multicols}
