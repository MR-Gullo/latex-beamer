\begin{multicols}{2}

%% EMERGING
\begin{tcolorbox}[profbox, colback=emerging!15, colframe=emerging!60,
  title={\colorbox{emerging!80}{\strut\hspace{2pt}EMERGING\hspace{2pt}}}]
\small\itshape Beginning awareness of special relativity concepts
\tcblower
\small
\begin{itemize}
\item Attempts basic calculations but makes significant algebraic or conceptual errors
\item Identifies the scenario but cannot explain different perspectives convincingly
\item Recognizes a paradox exists but cannot articulate why it appears contradictory
\item Submits work more than 3 days late, uses incorrect file formats, leaves work incomplete
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% DEVELOPING
\begin{tcolorbox}[profbox, colback=developing!15, colframe=developing!60,
  title={\colorbox{developing!80}{\strut\hspace{2pt}DEVELOPING\hspace{2pt}}}]
\small\itshape Basic grasp of special relativity, needs guidance for complex applications
\tcblower
\small
\begin{itemize}
\item Completes calculations with minor errors, sometimes plugging in numbers before showing algebra
\item Presents both reference frames but explanations remain surface level
\item States the paradox but struggles to fully explain the resolution
\item Submits 1--2 days late or requests last-minute extensions
\end{itemize}
\end{tcolorbox}

\columnbreak

%% PROFICIENT
\begin{tcolorbox}[profbox, colback=proficient!15, colframe=proficient!60,
  title={\colorbox{proficient!80}{\strut\hspace{2pt}PROFICIENT\hspace{2pt}}}]
\small\itshape Solid understanding, applies concepts independently
\tcblower
\small
\begin{itemize}
\item Performs calculations accurately, showing algebraic manipulation before plugging in values
\item Explains both perspectives convincingly with supporting physics reasoning
\item Clearly articulates paradox and resolves it using time dilation, length contraction, or simultaneity
\item Submits on time with proper formatting and communicates about delays
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% EXTENDING
\begin{tcolorbox}[profbox, colback=extending!15, colframe=extending!60,
  title={\colorbox{extending!80}{\strut\hspace{2pt}EXTENDING\hspace{2pt}}}]
\small\itshape Exceptional understanding with sophisticated connections
\tcblower
\small
\begin{itemize}
\item Executes calculations with full algebraic derivations, checks limiting cases, explains physical meaning
\item Synthesizes multiple relativity concepts showing consistent physical outcomes
\item Connects paradox resolution to experimental evidence and explores implications beyond requirements
\item Submits quality work ahead of deadlines with clear communication
\end{itemize}
\end{tcolorbox}
