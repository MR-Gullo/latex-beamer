\begin{multicols}{2}

%% EMERGING
\begin{tcolorbox}[profbox, colback=emerging!15, colframe=emerging!60,
  title={\colorbox{emerging!80}{\strut\hspace{2pt}EMERGING\hspace{2pt}}}]
\small\itshape Minimal effort towards timely submission and completion
\tcblower
\small
\begin{itemize}
\item Submits work more than 3 days late without communication
\item Uses incorrect file formats consistently
\item Leaves significant work incomplete across assignments
\item Shows little awareness of deadline management
\item Demonstrates minimal organizational systems
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% DEVELOPING
\begin{tcolorbox}[profbox, colback=developing!15, colframe=developing!60,
  title={\colorbox{developing!80}{\strut\hspace{2pt}DEVELOPING\hspace{2pt}}}]
\small\itshape Shows basic attention to deadlines but struggles with consistency
\tcblower
\small
\begin{itemize}
\item Submits 1--2 days late or requests last-minute extensions
\item Generally follows file format requirements with some errors
\item Completes most work but may rush through final components
\item Shows emerging awareness of time management needs
\item Begins to develop basic organizational approaches
\end{itemize}
\end{tcolorbox}

\columnbreak

%% PROFICIENT
\begin{tcolorbox}[profbox, colback=proficient!15, colframe=proficient!60,
  title={\colorbox{proficient!80}{\strut\hspace{2pt}PROFICIENT\hspace{2pt}}}]
\small\itshape Consistently reliable with submissions and communication
\tcblower
\small
\begin{itemize}
\item Submits on time with proper formatting across all assignments
\item Communicates proactively about potential delays when needed
\item Completes all components thoroughly and thoughtfully
\item Demonstrates solid time management and planning skills
\item Maintains consistent organizational systems
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% EXTENDING
\begin{tcolorbox}[profbox, colback=extending!15, colframe=extending!60,
  title={\colorbox{extending!80}{\strut\hspace{2pt}EXTENDING\hspace{2pt}}}]
\small\itshape Exceptional responsibility in managing work and deadlines
\tcblower
\small
\begin{itemize}
\item Submits quality work ahead of deadlines consistently
\item Maintains clear, proactive communication throughout projects
\item Prepares comprehensively for known absences and transitions
\item Creates and refines personal systems for tracking requirements
\item Models excellent organizational practices for others
\end{itemize}
\end{tcolorbox}
