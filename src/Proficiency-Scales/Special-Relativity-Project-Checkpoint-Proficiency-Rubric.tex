\begin{multicols}{2}

%% EMERGING
\begin{tcolorbox}[profbox, colback=emerging!15, colframe=emerging!60]
{\large\bfseries\sffamily EMERGING}\\[2pt]
\small\itshape Little progress; no clear plan; cannot explain topic or next steps
\tcblower
\small
\begin{itemize}
\item Dialogue not started or only has a few lines; no clear structure or character perspectives
\item Calculations not started or contain major errors; no algebra shown
\item No video planning; group has not discussed filming, visuals, or who does what
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% DEVELOPING
\begin{tcolorbox}[profbox, colback=developing!15, colframe=developing!60]
{\large\bfseries\sffamily DEVELOPING}\\[2pt]
\small\itshape Started work but behind schedule; key pieces missing; basic topic understanding
\tcblower
\small
\begin{itemize}
\item Dialogue outline exists; some scenes written but missing the paradox setup or resolution
\item Some calculations done but incomplete; algebra not fully shown or numbers plugged in too early
\item Basic video plan exists but missing details; roles assigned but diagrams or props not ready
\end{itemize}
\end{tcolorbox}

\columnbreak

%% PROFICIENT
\begin{tcolorbox}[profbox, colback=proficient!15, colframe=proficient!60]
{\large\bfseries\sffamily PROFICIENT}\\[2pt]
\small\itshape On track; dialogue mostly complete; clear video plan with roles assigned
\tcblower
\small
\begin{itemize}
\item Dialogue draft complete with both perspectives, paradox clearly stated, and resolution included
\item All required calculations done correctly with algebra shown first, then numbers at the end
\item Video plan covers all 4 segments; diagrams or props in progress; filming schedule set
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% EXTENDING
\begin{tcolorbox}[profbox, colback=extending!15, colframe=extending!60]
{\large\bfseries\sffamily EXTENDING}\\[2pt]
\small\itshape Ahead of schedule; high-quality work; deep understanding; helps others
\tcblower
\small
\begin{itemize}
\item Dialogue is polished with natural conversation, strong character voices, and clear physics explanations
\item Calculations go beyond requirements; group can extend to new scenarios or explain why formulas work
\item Video segments already filmed or ready to film; creative visuals, props, or demonstrations planned
\end{itemize}
\end{tcolorbox}

\end{multicols}
