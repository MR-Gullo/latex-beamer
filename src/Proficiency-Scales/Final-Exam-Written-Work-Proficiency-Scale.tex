%% EMERGING
\begin{tcolorbox}[profbox, colback=emerging!15, colframe=emerging!60,
  title={\colorbox{emerging!80}{\strut\hspace{2pt}EMERGING\hspace{2pt}}}]
\small\itshape Beginning to show your thinking across all questions
\tcblower
\small
\begin{itemize}
\item Lists some information randomly
\item Shows scattered work without clear order
\item Often forgets units and important details
\item Incomplete responses across problems
\item Attempts basic sketches when helpful
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% DEVELOPING
\begin{tcolorbox}[profbox, colback=developing!15, colframe=developing!60,
  title={\colorbox{developing!80}{\strut\hspace{2pt}DEVELOPING\hspace{2pt}}}]
\small\itshape Growing understanding of organized thinking
\tcblower
\small
\begin{itemize}
\item Organizes information with some gaps
\item Shows work in sequence most of the time
\item Uses units sometimes across responses
\item Addresses most parts of questions
\item Creates labeled sketches in some responses
\end{itemize}
\end{tcolorbox}

\columnbreak

%% PROFICIENT
\begin{tcolorbox}[profbox, colback=proficient!15, colframe=proficient!60,
  title={\colorbox{proficient!80}{\strut\hspace{2pt}PROFICIENT\hspace{2pt}}}]
\small\itshape Clear, organized thinking consistently shown
\tcblower
\small
\begin{itemize}
\item Lists all important information clearly
\item Shows logical steps throughout all responses
\item Uses units consistently across all responses
\item Addresses all parts of every question completely
\item Makes clear, labeled sketches consistently when helpful
\end{itemize}
\end{tcolorbox}

\vspace{6pt}

%% EXTENDING
\begin{tcolorbox}[profbox, colback=extending!15, colframe=extending!60,
  title={\colorbox{extending!80}{\strut\hspace{2pt}EXTENDING\hspace{2pt}}}]
\small\itshape Exceptional clarity and deep thinking throughout
\tcblower
\small
\begin{itemize}
\item Includes hidden details and assumptions
\item Plans approach before starting each response
\item Uses units consistently and checks them for accuracy
\item Connects ideas across different questions
\item Uses multiple sketches when helpful to clarify thinking
\end{itemize}
\end{tcolorbox}

\begin{multicols}{2}
\begin{tcolorbox}[colback=gray!10, colframe=gray!40, boxrule=0.5pt, arc=2pt]
\small\textbf{For Calculation Problems:}
\begin{itemize}
\item List givens and unknowns with units
\item Include labeled sketch when helpful
\item Show equation steps with units throughout
\item Clear final answer with correct units
\end{itemize}
\end{tcolorbox}

\columnbreak

\begin{tcolorbox}[colback=gray!10, colframe=gray!40, boxrule=0.5pt, arc=2pt]
\small\textbf{For Concept/Explanation Problems:}
\begin{itemize}
\item State what the question asks
\item Include sketch or diagram when helpful
\item Give reasoning steps in order
\item Connect to physics principles
\item Clear final answer with units when applicable
\end{itemize}
\end{tcolorbox}
