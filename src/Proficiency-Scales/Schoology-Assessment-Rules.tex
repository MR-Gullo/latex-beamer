\large
\begin{enumerate}
\item \textbf{ONLY ONE} application open: \textbf{WEB BROWSER}
\item Only \textbf{ONE TAB OPEN}: Schoology assessment page
\item Do \textbf{NOT} use fullscreen mode
\item \textbf{NO} adjacent desktops open
\item Additional resources allowed \textbf{ONLY IF SPECIFIED}
\item Screen \textbf{BRIGHTNESS} at \textbf{MAXIMUM}
\item \textbf{DEVICE ON DESK}, \textbf{SQUARE} to the desk's edge
\item \textbf{DON'T HIGHLIGHT} any text
\item \textbf{TURN OFF} Grammarly or any writing assist software
\item These rules \textbf{APPLY TO ALL} teachers' assessments
\end{enumerate}

\begin{tcolorbox}[colback=notecolor!15, colframe=notecolor!60, boxrule=0.5pt, arc=2pt, title={\bfseries\sffamily Computer Lab Setup}]
\begin{itemize}[nosep, itemsep=3pt]
\item Have students turn in their cell phones/watches
\item Boot up all computers (Ctrl+A then power on)
\item Power button is a drop-down ``action!'' looking icon at the top
\item \textbf{Step 1:} OBS at the bottom to start recording
\item \textbf{Step 2:} Open Schoology for students
\end{itemize}
