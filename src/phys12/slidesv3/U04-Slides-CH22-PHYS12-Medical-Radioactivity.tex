\begin{frame}
\frametitle{22.5 The Physics of PET}
\textbf{Positron-electron annihilation:}

\pause
\vspace{0.3cm}

\begin{enumerate}
\item Radiopharmaceutical emits positron ($e^+$) \pause
\item Positron encounters electron ($e^-$) in tissue \pause
\item \textbf{Annihilation:} Both particles destroyed \pause
\item Two gamma rays created, each with 0.511 MeV \energy{energy} \pause
\item Gamma rays travel in \textbf{exactly opposite directions}
\end{enumerate}

\pause
\vspace{0.3cm}

\begin{block}{The Universal Law}
\begin{center}
$\energy{E} = \mass{m}c^2$
\end{center}
\mass{Mass} converts to \energy{energy}: electron \mass{mass} = 0.511 MeV per gamma ray.
\end{block}
\note{[P0] "Positron-electron annihilation - the core of PET"\\\\
[P1] "Radiopharmaceutical emits positron - antimatter"\\\\
[P2] "Positron travels short distance, encounters electron"\\\\
[P3] "Annihilation: both particles destroyed completely"\\\\
[P4] "Two gamma rays created from mass-energy conversion"\\\\
[P5] "Gamma rays travel exactly opposite directions - conservation of momentum"\\\\
[P6] [THE REVELATION] "Einstein's E equals m c squared in action"\\\\
[ALGEBRA] "Electron mass 0.511 MeV, two photons, so 0.511 MeV each"\\\\
[THE WONDER] Mass becomes pure energy - we detect it to save lives}
\end{frame}

\section{Radiation Effects on Cells}

\begin{frame}
\frametitle{22.5 The Double-Edged Sword}
\begin{alertblock}{The Paradox}
\textbf{Ionizing radiation:}
\begin{itemize}
\item Can cause cancer, burns, and hair loss
\item Can treat and cure cancer
\end{itemize}
How do we understand this?
\end{alertblock}

\pause
\vspace{0.3cm}

\begin{block}{The Universal Truth}
All effects of ionizing radiation come from one source:\\
\textbf{Damage to DNA molecules inside cells.}
\end{block}
\note{[P0] "Ionizing radiation: both curse and cure"\\\\
[P1] [THE CONFLICT] "How can same thing cause and cure cancer?"\\\\
[P2] [THE REVELATION] "All effects trace to one source: DNA damage"\\\\
- DNA is the instruction manual for cells\\\\
- Radiation breaks chemical bonds in DNA\\\\
- Cell's response to damage determines outcome\\\\
[THE HUMILITY] This confused scientists for decades\\\\
[THE WONDER] Understanding DNA unlocked radiation therapy}
\end{frame}

\begin{frame}
\frametitle{22.5 DNA: The Code of Life}
\textbf{DNA structure:}
\begin{itemize}
\item Long, double-helix molecules \pause
\item Contains genetic codes that govern cell function \pause
\item Every cell experiences up to 1 million DNA damage events per day \pause
\item DNA repair ability is vital for survival
\end{itemize}

\pause
\vspace{0.3cm}

\begin{exampleblock}{The Mental Model}
DNA is like the source code for your body. Radiation introduces bugs. The cell tries to debug.
\end{exampleblock}
\note{[P0] "DNA: double helix molecule containing genetic code"\\\\
[P1] "Genetic codes tell cell what to do, when to divide, when to die"\\\\
[P2] "Every cell: up to 1 million damage events per day - from metabolism, environment"\\\\
[P3] "DNA repair is vital - cells have molecular machines that fix breaks"\\\\
[P4] [THE CONNECTION - Digital Archetype] "DNA is source code, radiation introduces bugs"\\\\
[THE HUMILITY] "Most damage gets repaired. Sometimes repair fails."\\\\
[THE WONDER] Your cells are repairing DNA right now as you listen}
\end{frame}

\begin{frame}
\frametitle{22.5 When Repair Fails}
\textbf{A cell with damaged DNA repair ability can:}

\pause
\vspace{0.3cm}

\begin{enumerate}
\item \textbf{Senescence:} Enter permanent dormant state (retires) \pause
\item \textbf{Apoptosis:} Commit programmed suicide (cell death) \pause
\item \textbf{Unregulated division:} Become cancerous (tumor)
\end{enumerate}

\pause
\vspace{0.3cm}

\begin{alertblock}{Civilian View vs. Reality}
\textbf{Civilian:} "Radiation always causes cancer."\\
\textbf{Physicist:} "Radiation damages DNA. Outcome depends on dose, cell type, and repair ability."
\end{alertblock}
\note{[P0] "When DNA repair fails, cell has three fates"\\\\
[P1] "Senescence: cell stops dividing permanently - goes dormant"\\\\
[P2] "Apoptosis: programmed cell death - cell destroys itself cleanly"\\\\
[P3] "Unregulated division: damaged cell ignores stop signals - becomes tumor"\\\\
[P4] [THE CONFLICT] "Radiation doesn't always cause cancer - it damages DNA"\\\\
- Low doses: repair usually succeeds\\\\
- High doses: cell death (apoptosis) or cancer (failed repair)\\\\
[THE REVELATION] Cancer is uncontrolled cell division - radiation can cause or cure}
\end{frame}

\begin{frame}
\frametitle{22.5 Why Radiation Kills Cancer}
\begin{block}{The Vulnerability}
Cancer cells reproduce rapidly. Radiation damages DNA most during cell division.
\end{block}

\pause
\vspace{0.3cm}

\textbf{Cancer cells are MORE sensitive to radiation than normal cells.}

\pause
\vspace{0.3cm}

\begin{exampleblock}{The Strategy}
Target rapidly dividing cancer cells while minimizing damage to slowly dividing normal cells.
\end{exampleblock}

\pause
\vspace{0.3cm}

\alert{This is why radiation can both cause and cure cancer.}
\note{[P0] "Cancer characterized by rapid, uncontrolled cell division"\\\\
[P1] "Radiation damages DNA most during division - DNA is exposed, uncoiled"\\\\
[P2] "Cancer cells divide rapidly, so more vulnerable than normal cells"\\\\
[P3] [THE REVELATION] "Target the thing that makes cancer dangerous - rapid division"\\\\
[P4] [THE PARADOX RESOLVED] "Radiation causes cancer by damaging DNA. Radiation cures cancer by damaging DNA of dividing cells."\\\\
[THE WONDER] The dose makes the poison - and the cure}
\end{frame}

\section{Radiation Therapy}

\begin{frame}
\frametitle{22.5 The Central Challenge}
\begin{block}{Therapeutic Ratio}
$$\text{Therapeutic Ratio} = \frac{\text{Abnormal cells killed}}{\text{Normal cells killed}}$$
\end{block}

\pause
\vspace{0.3cm}

\textbf{Goal of radiotherapy:}
\begin{itemize}
\item Maximize dose to cancer cells \pause
\item Minimize dose to normal cells \pause
\item Achieve highest possible therapeutic ratio
\end{itemize}

\pause
\vspace{0.3cm}

\begin{alertblock}{The Trade-off}
Radiation doesn't distinguish cancer from normal cells - only rapidly dividing from slowly dividing.
\end{alertblock}
\note{[P0] "Therapeutic ratio: abnormal cells killed over normal cells killed"\\\\
[P1] "Maximize dose to cancer cells"\\\\
[P2] "Minimize dose to normal cells"\\\\
[P3] "Achieve highest possible ratio - kill cancer, spare healthy tissue"\\\\
[P4] [THE CONFLICT] "Radiation doesn't magically know which cells are bad"\\\\
- It kills rapidly dividing cells\\\\
- Cancer divides rapidly, but so do hair, intestinal lining, bone marrow\\\\
[THE HUMILITY] This is why chemotherapy causes hair loss and nausea\\\\
- Targeting is the engineering challenge}
\end{frame}

\section{Radiation Dosage}

\begin{frame}
\frametitle{22.5 Measuring Damage}
\textbf{To quantify biological effects, we need a dose unit.}

\pause
\vspace{0.3cm}

\begin{block}{The Rad (Radiation Absorbed Dose)}
$$1 \text{ rad} = 0.01 \text{ J/kg}$$
Amount of ionizing \energy{energy} deposited per kilogram of tissue.
\end{block}

\pause
\vspace{0.3cm}

\textbf{Key point:} Must specify which tissue received the dose.
\begin{itemize}
\item Whole-body dose
\item Localized dose (e.g., forearm)
\end{itemize}
\note{[P0] "To quantify biological effects, need dose unit"\\\\
[P1] "The rad: 1/100 of joule per kilogram of tissue"\\\\
[P2] "Must specify which tissue - whole body vs localized"\\\\
[ALGEBRA] "Rad equals 0.01 joules per kilogram"\\\\
- Small energy per kilogram, but huge effects\\\\
- Only few eV needed to ionize atom\\\\
- 0.01 J creates massive number of ion pairs\\\\
[THE REVELATION] Energy deposited determines damage potential}
\end{frame}

\begin{frame}
\frametitle{Attempt: Radiation Dose Calculation}
\begin{exampleblock}{The Challenge (3 min, silent)}
A 60.0-kg person is exposed to ionizing radiation over their entire body and absorbs 1.80 J of energy.

\vspace{0.3cm}

\textbf{Given:}
\begin{itemize}
\item \mass{Mass} = 60.0 kg
\item \energy{Energy} absorbed = 1.80 J
\end{itemize}

\textbf{Find:} Whole-body radiation dose in rads

\vspace{0.3cm}

\textit{Can you calculate the dose? Work silently.}
\end{exampleblock}
\note{[THE CHALLENGE] Can they calculate medical dose from energy?\\\\
[SAY] "Try this on your own. Unit conversion from joules per kg to rads."\\\\
[TIMING] 3-4 min SILENT individual work\\\\
[CIRCULATE] Note who converts correctly, who forgets the 0.01 factor\\\\
[WATCH FOR] Students dividing instead of multiplying by conversion\\\\
[DON'T HELP] Let them struggle with unit conversion - critical skill}
\end{frame}

\begin{frame}
\frametitle{Compare: Dose Calculation}
\textbf{Turn and talk (2 min):}

\vspace{0.3cm}

\begin{enumerate}
\item What formula relates \energy{energy}, \mass{mass}, and dose?
\item How do you convert J/kg to rads?
\item What conversion factor did you use?
\end{enumerate}

\vspace{0.5cm}

\pause
\alert{Name wheel:} One pair share your approach (not your answer).
\note{[TIMING] 2-3 min pair discussion\\\\
[CIRCULATE] Listen for common approaches\\\\
[CHECK] Name wheel: call a pair to share approach\\\\
[EXPECTED APPROACH] Divide energy by mass to get J/kg, then use 1 rad equals 0.01 J/kg\\\\
[COMMON ERROR] Multiplying by 0.01 instead of dividing}
\end{frame}

\begin{frame}
\frametitle{Reveal: The Dose}
\textbf{Self-correct in a different color:}

\vspace{0.3cm}

\textbf{Step 1:} \energy{Energy} per \mass{mass}
$$\frac{1.80 \text{ J}}{60.0 \text{ kg}} = 0.0300 \text{ J/kg}$$

\pause
\vspace{0.2cm}

\textbf{Step 2:} Convert using $1 \text{ rad} = 0.01 \text{ J/kg}$

\pause
\vspace{0.2cm}

$$\text{Dose} = \frac{0.0300 \text{ J/kg}}{0.01 \text{ J/kg per rad}} = \boxed{3.00 \text{ rad}}$$

\pause
\textbf{Check:} 1.80 J over 60 kg is small \energy{energy} per \mass{mass}, so low rad dose. Reasonable!
\note{[P0] "Self-correct in a different color"\\\\
[P1] [ALGEBRA] "Energy over mass: 1.80 J divided by 60 kg equals 0.03 J per kg"\\\\
[P2] "Convert using 1 rad equals 0.01 J per kg"\\\\
[P3] [ANSWER] "3.00 rad - whole-body dose"\\\\
- [ALTERNATIVE METHOD] Multiply J/kg by 100 to get rads\\\\
[THE WONDER] Just calculated a medical dose - this number determines treatment\\\\
[CHECK] Small energy, low dose - makes sense}
\end{frame}

\begin{frame}
\frametitle{22.5 Not All Radiation Is Equal}
\textbf{For same dose in rads, effects depend on:}

\pause
\vspace{0.3cm}

\begin{itemize}
\item Type of radiation ($\alpha$, $\beta$, $\gamma$, X-ray, neutron) \pause
\item Type of tissue (eyes, bone, muscle)
\end{itemize}

\pause
\vspace{0.3cm}

\begin{block}{Relative Biological Effectiveness (RBE)}
Number expressing relative damage from a given type of radiation.
\end{block}

\pause
\vspace{0.3cm}

\begin{itemize}
