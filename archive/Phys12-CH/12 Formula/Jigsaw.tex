\section*{What We Will Do Today}
\begin{center}
\fbox{\parbox{0.9\textwidth}{
\textbf{Part 1 :} Study chapter and make slides\\
\textbf{Part 2 :} Present to class
}}
\end{center}

\section*{Step 1: Choose Group Leaders (5 min)}
\begin{itemize}[leftmargin=*]
    \item Each group picks one new leader
    \item Leaders meet teacher at front
    \item Leaders split chapter parts fairly
\end{itemize}

\section*{Step 2: Study and Make Slides }

\subsection*{What to Put in Your Slides:}
\begin{center}
\fbox{\parbox{0.9\textwidth}{
\begin{enumerate}[leftmargin=*]
    \item Title slide (topic + names)
    \item Important formulas (take screenshots!)
    \item One worked example (from textbook)
    \item Online resource/media (not video)
    \item Summary of main ideas
\end{enumerate}
}}
\end{center}

\subsection*{Suggested Group Jobs:}
\begin{center}
\fbox{\parbox{0.9\textwidth}{
\begin{itemize}[leftmargin=*]
    \item \textbf{Leader:} Keeps time, checks work
    \item \textbf{Formula Person:} Takes formula pictures
    \item \textbf{Example Expert:} Explains sample problem
    \item \textbf{Resource Finder:} Finds online help
    \item \textbf{Slide Maker:} Creates presentation
\end{itemize}
}}
\end{center}

\section*{Step 3: Post Your Work}
\begin{center}
\fbox{\parbox{0.9\textwidth}{
Before break, leaders post to discussion board:
\begin{itemize}[leftmargin=*]
    \item Your slides
    \item Textbook Page numbers used
    \item Group member names and jobs
\end{itemize}
}}
\end{center}

\section*{Step 4: Present to Class}
\begin{center}
\fbox{\parbox{0.9\textwidth}{
\begin{itemize}[leftmargin=*]
    \item Each group gets 5-10 or 10-15 minutes
    \item Everyone must speak
    \item Explain your worked example
    \item Show your online resource
\end{itemize}
}}
\end{center}

\section*{Tips for Success}
\begin{center}
\fbox{\parbox{0.9\textwidth}{
\textbf{DO:}
\begin{itemize}[leftmargin=*]
    \item Take pictures of formulas
    \item Choose a simple example
    \item Practice your timing
    \item Post slides before break
\end{itemize}

\textbf{DON'T:}
\begin{itemize}[leftmargin=*]
    \item Type formulas by hand
    \item Pick hard examples
    \item Make text-heavy slides
    \item Talk more than 5 minutes
\end{itemize}
}}
\end{center}
